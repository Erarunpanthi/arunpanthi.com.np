Tab Opener Controller

// Function to open a new tab and close the previous one
let previousTab = null;

function openAndCloseTabs() {
    // Open a new tab
    const newTab = window.open('https://askvoid.com/?slice', '_blank');

    // Close the previous tab if it exists
    if (previousTab && !previousTab.closed) {
        previousTab.close();
    }

    // Update the previousTab reference to the newly opened tab
    previousTab = newTab;
}

// Set an interval to open and close tabs every 5 seconds
const intervalId = setInterval(openAndCloseTabs, 5000);

// Optionally, stop the interval after a certain number of iterations
// For example, stop after 10 iterations (10 * 5 seconds = 50 seconds)
let iterationCount = 0;
const maxIterations = 10;

function stopInterval() {
    if (iterationCount >= maxIterations) {
        clearInterval(intervalId);
        console.log('Stopped opening new tabs.');
    }
    iterationCount++;
}

// Call stopInterval within the openAndCloseTabs function
function openAndCloseTabsWithStop() {
    openAndCloseTabs();
    stopInterval();
}

// Update the interval to use the new function
clearInterval(intervalId);
const newIntervalId = setInterval(openAndCloseTabsWithStop, 8000);
